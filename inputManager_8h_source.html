439 // helper function to read from the translation config file to get an option for a particular sub
440 // node (e.g. Parameters or Samples) returns true and sets "ret" if the option is specified for this
441 // fitter and parameter otherwise returns false
442 template <typename T>
443 inline bool getFitterSpecificOption(const std::string &fitter, const std::string &option, T &ret, const std::string &parameter,
444 YAML::Node subConfig) const{
445 if (subConfig[parameter])
446 {
447
448 // EM: this is config definition of fitter specific names for this parameter
449 YAML::Node paramTranslation = subConfig[parameter];
450
451 if (paramTranslation[fitter])
452 {
453 // EM: then this is definition of how to find parameter in the specified fitter
454 YAML::Node fitterParamTranslation = paramTranslation[fitter];
455
456 if (fitterParamTranslation[option])
457 {
458 ret = fitterParamTranslation[option].as<T>();
459 return true;
460 }
461 }
462 }
463 return false;
464 }
